Create figureÂ¶

def gen_map(geodata, value, title):
    base = alt.Chart(geodata, title = title).mark_geoshape(
        stroke='black',
        strokeWidth=1
    )
    
    # Choropleth Layer
    choro = alt.Chart(geodata).mark_geoshape(
        fill='lightgray',
        stroke='black'
    ).encode(
        fill = alt.Color(value,  
                  scale=alt.Scale(scheme='greenblue'),
                  title = "Arbeidsledige")
    )
    
    # Labels Layer
    labels = alt.Chart(geodata).mark_text(
        baseline='top'
     ).encode(
         longitude='properties.centroid_lon:Q',
         latitude='properties.centroid_lat:Q',
         text= value,
         size=alt.value(8),
         opacity=alt.value(1)
     )
    
    fig = base + choro + labels

    return fig

ledige_2000_map = gen_map(geodata=no_data, value='properties.2000:Q', title='2000')
ledige_2007_map = gen_map(geodata=no_data, value='properties.2007:Q', title='2007')
ledige_2014_map = gen_map(geodata=no_data, value='properties.2014:Q', title='2014')

fig = alt.vconcat(ledige_2000_map,ledige_2007_map,ledige_2014_map)
