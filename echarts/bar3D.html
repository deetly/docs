Sample bar 3D charts

Source: pyecharts test

import deetly
import random

from pyecharts.charts import Bar3D
from pyecharts import options as opts
from pyecharts.globals import ChartType
from pyecharts.faker import Faker

# Create datapackage
metadata = {
    "name": "EChart Bar3D examples",
    "author": "deetly",
    "keywords": ["3D bar","barchart","chart","examples"]
}

dp = deetly.package(metadata)

data = [(i, j, random.randint(0, 12)) for i in range(6) for j in range(24)]
g = (
    Bar3D()
    .add(
        "",
        [[d[1], d[0], d[2]] for d in data],
        xaxis3d_opts=opts.Axis3DOpts(Faker.clock, type_="category"),
        yaxis3d_opts=opts.Axis3DOpts(Faker.week_en, type_="category"),
        zaxis3d_opts=opts.Axis3DOpts(type_="value"),
    )
    .set_global_opts(visualmap_opts=opts.VisualMapOpts(max_=20))
)

g.render_notebook()

dp.echart(g, "3D Bars sample")

data1 = [(i, j, random.randint(0, 12)) for i in range(6) for j in range(24)]
data2 = [(i, j, random.randint(13, 20)) for i in range(6) for j in range(24)]

g = (
    Bar3D()
    .add(
        "1",
        [[d[1], d[0], d[2]] for d in data1],
        xaxis3d_opts=opts.Axis3DOpts(Faker.clock, type_="category"),
        yaxis3d_opts=opts.Axis3DOpts(Faker.week_en, type_="category"),
        zaxis3d_opts=opts.Axis3DOpts(type_="value"),
    )
    .add(
        "2",
        [[d[1], d[0], d[2]] for d in data2],
        xaxis3d_opts=opts.Axis3DOpts(Faker.clock, type_="category"),
        yaxis3d_opts=opts.Axis3DOpts(Faker.week_en, type_="category"),
        zaxis3d_opts=opts.Axis3DOpts(type_="value"),
    )
    .set_global_opts(visualmap_opts=opts.VisualMapOpts(max_=20))
    .set_series_opts(**{"stack": "stack"})
)

g.render_notebook()

dp.echart(g, "Stacked 3D bar sample")
