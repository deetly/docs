data = df[df['Type']=='Country/Area'][['Region, subregion, country or area *','2020']].values.tolist()

high = max([x for _, x in data[1:]])
low = min([x for _, x in data[1:]])

from pyecharts.charts import Map
import pyecharts.options as opts

m = (Map().add(
    maptype='world',
    series_name="World Population",
    data_pair=data[1:],
    is_map_symbol_show=False,
    label_opts=opts.LabelOpts(is_show=False),
))
m.render_notebook()

m.set_global_opts(
    visualmap_opts=opts.VisualMapOpts(
        min_=low,
        max_=high,
        range_text=["max", "min"],
        is_calculable=True,
        range_color=["lightskyblue", "yellow", "orangered"],
    )
)
m.render_notebook()
