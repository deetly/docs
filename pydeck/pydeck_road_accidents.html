import os
import pydeck
import requests

import deetly

MAPBOX_TOKEN = os.environ['MAPBOX_TOKEN'] 

# Create datastory
package = {
    "name":"Example: UK Road accidents", 
    "keywords": ["example, pydeck, deck.gl, deck"]
}

ds = deetly.story(package)

DATA_URL = "https://raw.githubusercontent.com/uber-common/deck.gl-data/master/examples/geojson/vancouver-blocks.json"
data_geojson = requests.get(DATA_URL).json()

INITIAL_VIEW_STATE = pydeck.ViewState(
  latitude=49.254,
  longitude=-123.13,
  zoom=11,
  max_zoom=16,
)

gg = pydeck.Layer(
    'GeoJsonLayer',
    data_geojson,
    stroked=False,
    filled=True,
    extruded=True,
    wireframe=True,
    pickable=True,
    get_line_color=[255, 255, 255]

)

fig = pydeck.Deck(
    layers=[gg],
    initial_view_state=INITIAL_VIEW_STATE)

fig.show()

UK_ACCIDENTS_DATA = ('https://raw.githubusercontent.com/uber-common/deck.gl-data/master/examples/3d-heatmap/heatmap-data.csv')

layer = pydeck.Layer(
     'HexagonLayer',
     UK_ACCIDENTS_DATA,
     get_position=['lng', 'lat'],
     auto_highlight=True,
     elevation_scale=50,
     pickable=True,
     elevation_range=[0, 3000],
     extruded=True,
     coverage=1)

# Set the viewport location
view_state = pydeck.ViewState(
    longitude=-1.415,
    latitude=52.2323,
    zoom=6,
    min_zoom=5,
    max_zoom=15,
    pitch=40.5,
    bearing=-27.36)

# Render
fig = pydeck.Deck(layers=[layer], initial_view_state=view_state, api_keys={"mapbox": MAPBOX_TOKEN})
fig.show()

title = 'Road accidents'
description = 'Sample from deck documentation'

ds.pydeck(fig,title, description)

# custom tooltip
tooltip={
    'html': '<b>Count:</b> {elevationValue}',
    'style': {
        'color': 'white',
        'backgroundColor':'red',
        'zIndex': 1000
    }
}

fig = pydeck.Deck(
    layers=[layer],
    initial_view_state=view_state,
    api_keys={"mapbox": MAPBOX_TOKEN},
    tooltip = tooltip
)
fig.show()

title = 'Road accidents, custom tooltip'
description = 'Sample from deck documentation'

ds.pydeck(fig,title, description,tooltip=json.dumps(tooltip))
